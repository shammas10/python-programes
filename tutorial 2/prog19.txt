def fibonacci(n):
 
    if n <= 0:
        return 0
    elif n == 1:
        return 1
  
    else:
        return fibonacci(n - 1) + fibonacci(n - 2)

n = int(input("Enter the value of n to find the n-th Fibonacci number: "))

result = fibonacci(n)

print(f"The {n}-th Fibonacci number is: {result}")
